\let\AddBaseWitchClassFeatures\AddWitchClassFeatures
\renewcommand{\AddWitchClassFeatures}[1]{
	\AddBaseWitchClassFeatures{#1}
	\ifthenelse{#1 > 1}{
		\AddHex{Jinx-Eating}
		\AddConditionalSkillMod{+\DoMath{#1 / 2} to spellcraft checks to identify spells cast on you}
		\AddLimitedUse{Jinx-Eating}{\DoMath{\value{CharIntModCount} + 3}}
	\ifthenelse{#1 > 5}{
		\AddHex{Jinx-Gorging}
	\ifthenelse{#1 > 9}{
		\AddHex{Jinx-Feast}
	}{}}{}}{}	
}
