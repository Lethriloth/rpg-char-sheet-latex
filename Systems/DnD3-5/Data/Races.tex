\def\AddGreyElfRaceFeatures{
	\renewcommand\charrace{Grey Elf}
	\mediumsize
	\Addtocounter{charstrbasecount}{-2}
	\Addtocounter{chardexbasecount}{2}
	\Addtocounter{charconbasecount}{-2}
	\Addtocounter{charintbasecount}{2}
	\charspeed{30}
	\AddImmunity{Sleep}\addconditionaldefmod{+2 vs. Enchantment}
	\Addracefeature{Low-Light Vision}
	\Addracefeature{Elven Weapon Proficiencies}
	\Addracefeature{Keen Senses}
		\Addconditionalskillmod{Free Search check for secret doors within 5 ft.}
		\miscbonus[racial]{listen}{2}
		\miscbonus[racial]{search}{2}
		\miscbonus[racial]{spot}{2}
	\Addcharlanguage{Common}
	\Addcharlanguage{Elven}
}
