		}

		}{
			\AddSpellsknown{5}{(##1 - 7)/2}
			\AddSpellsknown{6}{(##1 - 9)/2}
		}
		\ifthenelse{##1 > 16}{\AddSpellsknown{7}{3}}{\AddSpellsknown{7}{(##1 - 11)/2}}
		\AddSpellsknown{8}{(##1 - 13) / 2}
		\AddSpellsknown{9}{##1 - 17}
	}
